% Fatos
pai(joao, maria).
pai(joao, pedro).
pai(carlos, joao).

mae(ana, maria).
mae(ana, pedro).

% Regra: X é avô de Y se X é pai do pai de Y
avo(X, Y) :-
    pai(X, Z),
    pai(Z, Y).

% Regra: X é irmão de Y se têm o mesmo pai e não são a mesma pessoa
irmao(X, Y) :-
    pai(P, X),
    pai(P, Y),
    X \= Y.

% Predicado para imprimir todos os avôs de maria
print_avos :-
    forall(avo(X, maria), (write('Avo de maria: '), write(X), nl)).

% Predicado para imprimir todos os irmãos de maria
print_irmaos :-
    forall(irmao(maria, Y), (write('Irmao de maria: '), write(Y), nl)).

% Executa os prints ao carregar o arquivo
:- print_avos.
:- print_irmaos.
:- halt.
